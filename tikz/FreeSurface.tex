\begin{tikzpicture}[scale=1.5]
  % Draw the wobbly free surface
  \fill[gray!20] % Fill below surface
  plot[domain=-3.5:3.5, samples=100, smooth] 
  (\x, {0.2*sin(3*\x r) + 0.2}) -- 
  (3.5,-1.5) -- (-3.5,-1.5) -- cycle;
  \draw[very thick, smooth, domain=-3.5:3.5, samples=100] 
  plot(\x, {0.2*sin(3*\x r) + 0.2}) node[right] {$f(\bfx,t)=0$};
  

  % Draw a normal vector pointing upwards
  \def\xpos{0.2};
  \filldraw[black] (\xpos, {0.2*sin(3*\xpos r) + 0.2}) circle (2pt);
  \node[below] at (\xpos, {0.2*sin(3*\xpos r) + 0.2}) {$\bfx$};
  \draw[->, thick] (\xpos, {0.2*sin(3*\xpos r) + 0.2}) -- ({\xpos-0.6*cos(3*\xpos r)}, {0.2*sin(3*\xpos r) + 0.2 +1}) 
  node[above] {$\bfn=\frac{\nabla f}{|\nabla f|}$}; % Normal vector
  \draw[->, thick] (\xpos, {0.2*sin(3*\xpos r) + 0.2}) -- ({\xpos+1}, {0.2*sin(3*\xpos r) + 0.2 + 0.6*cos(3*\xpos r)}) 
  node[above] {$\mathbf{t}$}; % Tangent vector
  \node[anchor=west] at (3.5,-1) {$f<0$};
  \node[anchor=west] at (3.5,1) {$f>0$};

  \node[anchor=west] at (-3.5,1) {$\bfS^+$, $p^+$};
  \node[anchor=west] at (-3.5,-1) {$\bfS^-$, $p^-$};
\end{tikzpicture}
